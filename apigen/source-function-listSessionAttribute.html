 75:     else if ($_SERVER['HTTP_X_FORWARDED'] != '127.0.0.1')
 76:         $ipaddress = $_SERVER['HTTP_X_FORWARDED'];
 77:     else if ($_SERVER['HTTP_FORWARDED_FOR'] != '127.0.0.1')
 78:         $ipaddress = $_SERVER['HTTP_FORWARDED_FOR'];
 79:     else if ($_SERVER['HTTP_FORWARDED'] != '127.0.0.1')
 80:         $ipaddress = $_SERVER['HTTP_FORWARDED'];
 81:     else if ($_SERVER['REMOTE_ADDR'] != '127.0.0.1')
 82:         $ipaddress = $_SERVER['REMOTE_ADDR'];
 83:     else
 84:         $ipaddress = 'UNKNOWN';
 85: 
 86:     return $ipaddress;
 87:     //Just get the headers if we can or else use the SERVER global
 88:     if (function_exists('apache_request_headers')) {
 89: 
 90:         $headers = apache_request_headers();
 91:     } else {
 92: 
 93:         $headers = $_SERVER;
 94:     }
 95: 
 96:     //Get the forwarded IP if it exists
 97:     if (array_key_exists('X-Forwarded-For', $headers) && filter_var($headers['X-Forwarded-For'], FILTER_VALIDATE_IP, FILTER_FLAG_IPV4)) {
 98: 
 99:         $the_ip = $headers['X-Forwarded-For'];
100:     } elseif (array_key_exists('HTTP_X_FORWARDED_FOR', $headers) && filter_var($headers['HTTP_X_FORWARDED_FOR'], FILTER_VALIDATE_IP, FILTER_FLAG_IPV4)
101:     ) {
102: 
103:         $the_ip = $headers['HTTP_X_FORWARDED_FOR'];
104:     } else {

106:         $the_ip = filter_var($_SERVER['REMOTE_ADDR'], FILTER_VALIDATE_IP, FILTER_FLAG_IPV4);
107:     }
